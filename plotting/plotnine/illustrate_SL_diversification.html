np.random.seed(5) # seed  5 is good

sd1 = sd2 = 10
cession = sd1 # * 1.65
num_losses = 300 # 250
gross_loss_1 = np.random.normal(0, sd1, size=num_losses)
gross_loss_2 = np.random.normal(0, sd2, size=num_losses)

gross_loss_1 = gross_loss_1[:, np.newaxis]
gross_loss_2 = gross_loss_2[:, np.newaxis]

losses = pd.DataFrame(np.concatenate((gross_loss_1, gross_loss_2), axis=1), columns=['loss1_gross', 'loss2_gross']) # np.shape([gross_loss_1, gross_loss_2])
losses['total_gross'] = losses.sum(axis=1)
losses['gross_prod'] = losses.loss1_gross * losses.loss2_gross
losses['gross_prod'] = losses['gross_prod'] / sd1*sd2 # np.sum(np.abs(losses['gross_prod'] ))

# calculate cession / net loss:
calc_cession = lambda x: x - cession if x > cession else 0
calc_net = lambda x: cession if x > cession else x
losses['total_net'] = [calc_net(x) for x in losses.total_gross]  # losses.assign(cession = ...)

# calculate direction of contribution to correlation
losses['prod_sign'] = [1 if x > 0 else -1 for x in losses.gross_prod]
losses.prod_sign = losses.prod_sign.astype('category') # or use 'factor(prod_sign)' in p9.aes

losses['total_gross_abs'] = np.abs(losses['total_gross'])
losses['total_net_abs'] = np.abs(losses['total_net'])

# lazy evaluation
# map(calc_cession, losses.total_gross.values
# print(list(xx))

print(losses.sum(axis=0))
print(np.corrcoef(losses[['loss1_gross', 'loss2_gross']].transpose()))
